INT fibo(x : INT){
   IF ( x < 1 ) THEN {
      RETURN -1;
   }
   IF ( x == 1 ) THEN {
      RETURN 1;
   }
   IF ( x == 2 ) THEN {
      RETURN 1;
   }
   RETURN fibo(x - 1) + fibo(x - 2);
}
